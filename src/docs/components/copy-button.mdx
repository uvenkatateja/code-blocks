## Usage

```tsx {1,4}
import { CopyButton } from "@/components/code-block/copy-button";

const Example = () => {
  return <CopyButton content="Text to be copied" />;
};

export default Example;
```

Or with ``<CodeBlock>`` component:

```tsx {1,8}
import { CopyButton } from "@/components/code-block/copy-button";
import { CodeBlock, CodeBlockHeader, CodeBlockContent } from "@/components/code-block/code-block";

const Example = () => {
  return (
    <CodeBlock>
      <CodeBlockHeader title="Example" language="javascript">
        <CopyButton content={`console.log('Hello, World!');`} />
      </CodeBlockHeader>
      <CodeBlockContent>
        {/* Your syntax highlighted code goes here. */}
      </CodeBlockContent>
    </CodeBlock>
  );
};
export default Example;
```
